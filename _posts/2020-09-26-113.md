思路：感觉挺简单的，一次过。dfs(二叉树遍历)+回溯

~~~c++
class Solution {
public:
    vector<vector<int>> ans;
    vector<int> temp;
    int cnt = 0;			//记录当前和
    void dfs(TreeNode* root, int sum){
        if(root == NULL)
            return;
        cnt += root->val;
        temp.push_back(root->val);
        if(root->left == NULL && root->right == NULL)		//如果是叶子节点就比较
            if(cnt == sum)
                ans.push_back(temp);
        dfs(root->left,sum);
        dfs(root->right,sum);
        cnt -= root->val;			//回溯
        temp.pop_back();
        return;
    }
    vector<vector<int>> pathSum(TreeNode* root, int sum) {
        if(root != NULL)
            dfs(root,sum);
        return ans;
    }
};
~~~
